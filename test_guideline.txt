<script>
    document.addEventListener('DOMContentLoaded', function () {
        // Get all tab buttons and input type fields
        const fileTabs = document.querySelectorAll('.nav-link');
        const inputTypeFile = document.getElementById('input_type_file');
        const inputTypeUrl = document.getElementById('input_type_url');
        const inputTypeText = document.getElementById('input_type_text');

        // Add event listeners to tabs
        fileTabs.forEach(tab => {
            tab.addEventListener('click', function () {
                // Disable all input type fields
                inputTypeFile.disabled = true;
                inputTypeUrl.disabled = true;
                inputTypeText.disabled = true;

                // Enable the selected input type field
                if (this.id === 'file-tab') {
                    inputTypeFile.disabled = false;
                } else if (this.id === 'url-tab') {
                    inputTypeUrl.disabled = false;
                } else if (this.id === 'text-tab') {
                    inputTypeText.disabled = false;
                }
            });
        });

        // Form validation
        const form = document.getElementById('guidelineForm');
        form.addEventListener('submit', function (event) {
            let isValid = true;
            const activeTab = document.querySelector('.nav-link.active').id;

            if (activeTab === 'file-tab') {
                const fileInput = document.getElementById('guidelines_file');
                if (!fileInput.files || fileInput.files.length === 0) {
                    alert('Please select a file to upload.');
                    isValid = false;
                }
            } else if (activeTab === 'url-tab') {
                const urlInput = document.getElementById('guidelines_url');
                if (!urlInput.value.trim()) {
                    alert('Please enter a URL.');
                    isValid = false;
                }
            } else if (activeTab === 'text-tab') {
                const textInput = document.getElementById('guidelines_text');
                if (!textInput.value.trim()) {
                    alert('Please enter some text.');
                    isValid = false;
                }
            }

            if (!isValid) {
                event.preventDefault();
            }
        });
    });
</script>
